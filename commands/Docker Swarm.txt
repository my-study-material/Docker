DOCKER SWARM SERVICE
====================
 ~ docker service create --name fun-service --replicas 3 -p 80:80 httpd
 ~ docker service create --replicas=5 web
 ~ docker service create --mode=global web
 ~ docker service create --mode=replicated --replicas=2 webapp
 ~ docker service scale webapp=4
 ~ docker service ls
 ~ docker service ps <service_name>
 ~ docker service inspect <service_name> --pretty
 ~ docker service logs <service_name>
 ~ docker service update -p 80:80 --image mysql:latest <service_name or id> 
 ~ docker service update -p 80:80 --update-delay 60s --image mysql:latest <service_name or id>
 ~ docker service update -p 80:80 --update-parallelism 3 --image mysql:latest <service_name or id>
 ~ docker service update -p 80:80 --update-failure-action pause|continue|rollback --image mysql:latest <service_name or id>
